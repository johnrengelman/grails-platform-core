001    package org.grails.plugin.platform.events.publisher;
002    
003    import org.apache.log4j.Logger;
004    import org.grails.plugin.platform.events.EventMessage;
005    import org.grails.plugin.platform.events.EventReply;
006    import org.grails.plugin.platform.events.dispatcher.GormTopicSupport;
007    import org.springframework.context.ApplicationEvent;
008    import org.springframework.context.ApplicationListener;
009    import org.springframework.util.ReflectionUtils;
010    
011    /**
012     * @author Stephane Maldini <smaldini@doc4web.com>
013     * @version 1.0
014     * @file
015     * @date 29/05/12
016     * @section DESCRIPTION
017     * <p/>
018     * [Does stuff]
019     */
020    public class GormBridgePublisher implements ApplicationListener {
021    
022        private GormTopicSupport gormTopicSupport;
023        private EventsPublisher grailsEventsPublisher;
024    
025        private final static Logger log = Logger.getLogger(GormBridgePublisher.class);
026        static final private String GORM_EVENT_PACKAGE = "org.grails.datastore.mapping.engine.event";
027    
028        public void setGrailsEventsPublisher(EventsPublisher grailsEventsPublisher) {
029            this.grailsEventsPublisher = grailsEventsPublisher;
030        }
031    
032        public void setGormTopicSupport(GormTopicSupport gormTopicSupport) {
033            this.gormTopicSupport = gormTopicSupport;
034        }
035    
036        public void onApplicationEvent(ApplicationEvent applicationEvent) {
037            //fixme horrible hack to support grails 1.3.x
038            if (applicationEvent.getClass().getName().startsWith(GORM_EVENT_PACKAGE)) {
039                String topic = gormTopicSupport.convertTopic(applicationEvent);
040                log.debug("sending " + applicationEvent + " to topic " + topic);
041                EventReply reply = grailsEventsPublisher.event(new EventMessage(topic,
042                        ReflectionUtils.invokeMethod(
043                                ReflectionUtils.findMethod(applicationEvent.getClass(), "getEntityObject"),
044                                applicationEvent
045                        ), GormTopicSupport.GORM_SOURCE, false));
046                try {
047                    gormTopicSupport.processCancel(applicationEvent, reply.getValues());
048                } catch (Exception e) {
049                    throw new RuntimeException(e);//shouldn't happen as its sync event
050                }
051            }
052        }
053    }
